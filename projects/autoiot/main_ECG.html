def load_ecg_records_from_database(database_path):
    """
    Load ECG records and their annotations from a given database path using the wfdb package.
    
    Parameters:
    - database_path: A string indicating the path to the database directory containing the ECG records.
    
    Returns:
    - A dictionary with record names as keys and tuples (ECG signal, sampling rate, annotations) as values.
    
    Explanations of the code:
    - The function iterates over all files in the given database directory.
    - For each .hea file found, it uses wfdb.rdrecord to read the ECG signal and wfdb.rdann to read annotations.
    - The function ensures the integrity of the data by catching possible exceptions during the file reading process.
    - The ECG signal, sampling rate, and annotations are stored in a dictionary for easy access.
    - In case of an error, an informative message is printed, and the function returns None.
    """
    
    ecg_records = {}
    
    try:
        # Iterate through each file in the database directory
        for file in os.listdir(database_path):
            if file.endswith('.hea'):
                record_name = file.rstrip('.hea')
                # Read the ECG record and annotation
                record = wfdb.rdrecord(os.path.join(database_path, record_name))
                annotation = wfdb.rdann(os.path.join(database_path, record_name), 'atr')
                # Store the ECG signal, sampling rate, and annotations in the dictionary
                ecg_records[record_name] = (record.p_signal, record.fs, annotation)
    
    except FileNotFoundError as fnf_error:
        print(f"Error: {fnf_error}")
        print("Please check the database path and ensure it contains the MIT-BIH Arrhythmia Database files.")
        return None
    except Exception as e:
        print(f"An unexpected error occurred: {e}")
        return None

    return ecg_records

import numpy as np

def evaluate_detection_results(detected_peaks, annotation):
    """
    Evaluate the detection results using standard evaluation metrics.

    Parameters:
    - detected_peaks: A numpy array of detected R-peak indices from the detection algorithm.
    - annotation: The wfdb.Annotation object containing the gold standard R-peak annotations.

    Returns:
    - A tuple containing the detection accuracy, sensitivity, PPV (positive predictive value), and F1 score.

    Explanations of the code:
    - The function matches detected peaks to the true annotations within a tolerance window.
    - True positives (TP) are detected peaks that match the annotations.
    - False positives (FP) are detected peaks that don't match any annotation.
    - False negatives (FN) are annotations that don't have a corresponding detected peak.
    - Sensitivity (also known as recall) is calculated as TP / (TP + FN).
    - PPV (also known as precision) is calculated as TP / (TP + FP).
    - F1 score is the harmonic mean of precision and recall.
    - Detection accuracy is the proportion of true results (both true positives and true negatives) among the total number of cases examined.
    - The function handles edge cases where no peaks are detected or no annotations are provided.
    """
    
    # Convert annotations to numpy array for efficient computation
    annotation_indices = np.array(annotation.sample)
    
    # Define a tolerance window in samples (e.g., 10 samples for a 360 Hz sampling rate corresponds to ~28 ms)
    tolerance = int(0.1 * annotation.fs)
    
    # Initialize counts for TP, FP, and FN
    TP = FP = FN = 0
    
    # Check each detected peak to determine if it is a TP or FP
    for peak in detected_peaks:
        if np.any(np.abs(annotation_indices - peak) <= tolerance):
            TP += 1
        else:
            FP += 1
    
    # Check each annotation to determine if it is a FN
    for ann_idx in annotation_indices:
        if not np.any(np.abs(detected_peaks - ann_idx) <= tolerance):
            FN += 1
    
    # Calculate standard evaluation metrics
    sensitivity = TP / (TP + FN) if TP + FN > 0 else 0
    PPV = TP / (TP + FP) if TP + FP > 0 else 0
    F1_score = 2 * (PPV * sensitivity) / (PPV + sensitivity) if PPV + sensitivity > 0 else 0
    accuracy = TP / len(annotation_indices) if len(annotation_indices) > 0 else 0
    
    return accuracy, sensitivity, PPV, F1_score

def preprocess_ecg_signals(ecg_records, desired_fs=360):
    """
    Preprocess the ECG records to a uniform sampling rate and apply a bandpass filter.

    Parameters:
    - ecg_records: dict, ECG records with record names as keys and tuples (signal, sampling_rate, annotations) as values.
    - desired_fs: int, the desired sampling rate to standardize the ECG signals.

    Returns:
    - preprocessed_records: dict, preprocessed ECG records with the same structure as ecg_records.

    Explanations of the code:
    - For each ECG record, resample the signal if the original sampling rate differs from the desired_fs.
    - Design a bandpass Butterworth filter within the typical ECG signal frequency band (0.5 Hz to 100 Hz).
    - Apply the filter to the signal using filtfilt to avoid phase distortion.
    - The preprocessed signal is stored back in the dictionary, maintaining the original keys.
    - Handles edge cases where the signal cannot be resampled or filtered due to data issues.
    """
    
    preprocessed_records = {}
    for record_name, (signal, fs, annotations) in ecg_records.items():
        try:
            # Resample signal if necessary
            if fs != desired_fs:
                num_samples = int(len(signal) * desired_fs / fs)
                signal = resample(signal, num_samples)
            
            # Bandpass filter design parameters
            lowcut = 0.5  # Low frequency cutoff for ECG signals in Hz
            highcut = 100  # High frequency cutoff for ECG signals in Hz
            nyquist = 0.5 * desired_fs
            low = lowcut / nyquist
            high = highcut / nyquist
            
            # Design the bandpass filter
            b, a = butter(1, [low, high], btype='band')
            
            # Apply the filter using filtfilt
            filtered_signal = filtfilt(b, a, signal)
            
            # Store preprocessed signal
            preprocessed_records[record_name] = (filtered_signal, desired_fs, annotations)
        except Exception as e:
            print(f"An error occurred while preprocessing record {record_name}: {e}")
    
    return preprocessed_records

def preprocess_ecg_signal(ecg_signal, fs, desired_fs=360):
    if fs != desired_fs:
        num_samples = int(len(ecg_signal) * desired_fs / fs)
        ecg_signal = resample(ecg_signal, num_samples)
    lowcut = 0.5
    highcut = 100
    nyquist = 0.5 * desired_fs
    low = lowcut / nyquist
    high = highcut / nyquist
    b, a = butter(1, [low, high], btype='band')
    filtered_ecg_signal = filtfilt(b, a, ecg_signal)
    return filtered_ecg_signal

def detect_r_peaks(ecg_signal, sampling_rate):
    diff_signal = np.diff(ecg_signal)
    squared_signal = diff_signal ** 2
    window_size = int(0.12 * sampling_rate)
    integrated_signal = np.convolve(squared_signal, np.ones(window_size)/window_size, mode='same')
    refractory_period = int(0.2 * sampling_rate)
    peaks, _ = find_peaks(integrated_signal, distance=refractory_period)
    return peaks

def evaluate_detection_results(detected_peaks, annotations):
    true_positives = []
    false_positives = []
    false_negatives = []
    tolerance_window = int(0.1 * annotations.fs)
    annotation_indices = np.array(annotations.sample)
    for peak_index in detected_peaks:
        closest_annotation = np.argmin(np.abs(annotation_indices - peak_index))
        if np.abs(annotation_indices[closest_annotation] - peak_index) <= tolerance_window:
            true_positives.append(peak_index)
        else:
            false_positives.append(peak_index)
    for annotation_index in annotation_indices:
        if not np.any(np.abs(detected_peaks - annotation_index) <= tolerance_window):
            false_negatives.append(annotation_index)
    accuracy = len(true_positives) / len(annotation_indices) if len(annotation_indices) > 0 else 0
    return accuracy, true_positives, false_positives, false_negatives

    r_peak_indices_list = detected_peaks.tolist()
    formatted_result = (
        f'Case {record_name}',
        f'R-peak indices: {np.array(r_peak_indices_list)}',
        f'Detection accuracy: {accuracy}'
    )
    return formatted_result
